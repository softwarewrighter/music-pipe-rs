use anyhow::{Context, Result};
use serde::{Deserialize, Serialize};
use std::io::{self, BufRead, Write};

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "type")]
pub enum Event {
    NoteOn { t: u32, ch: u8, key: u8, vel: u8 },
    NoteOff { t: u32, ch: u8, key: u8 },
    Tempo { t: u32, bpm: u32 },
    End { t: u32 },
}

pub fn read_events_from_stdin() -> Result<Vec<Event>> {
    let stdin = io::stdin();
    let mut out = Vec::new();
    for (i, line) in stdin.lock().lines().enumerate() {
        let line = line.context("stdin read failed")?;
        if line.trim().is_empty() {
            continue;
        }
        let ev: Event = serde_json::from_str(&line)
            .with_context(|| format!("invalid JSON on line {}", i + 1))?;
        out.push(ev);
    }
    Ok(out)
}

pub fn write_events_to_stdout(events: &[Event]) -> Result<()> {
    let stdout = io::stdout();
    let mut w = io::BufWriter::new(stdout.lock());
    for ev in events {
        serde_json::to_writer(&mut w, ev)?;
        w.write_all(b"\n")?;
    }
    w.flush()?;
    Ok(())
}
This reads the whole stream into memory for MVP simplicity. Later you can stream-transform line-by-line.

use anyhow::Result;
use clap::Parser;
use music_ir::{write_events_to_stdout, Event};

#[derive(Parser, Debug)]
struct Args {
    /// Base MIDI note number (e.g., 60 = middle C)
    #[arg(long, default_value_t = 60)]
    base: u8,

    /// Channel 0-15
    #[arg(long, default_value_t = 0)]
    ch: u8,

    /// Ticks per quarter note used by downstream; here only used for defaults
    #[arg(long, default_value_t = 480)]
    tpq: u32,

    /// BPM emitted as a Tempo event at t=0
    #[arg(long, default_value_t = 120)]
    bpm: u32,
}

fn main() -> Result<()> {
    let a = Args::parse();

    // Simple arpeggio motif: C E G C'
    let keys = [a.base, a.base + 4, a.base + 7, a.base + 12];
    let dur = a.tpq / 2; // eighth note
    let vel = 96u8;

    let mut events = Vec::new();
    events.push(Event::Tempo { t: 0, bpm: a.bpm });

    let mut t = 0u32;
    for &k in &keys {
        events.push(Event::NoteOn { t, ch: a.ch, key: k, vel });
        events.push(Event::NoteOff { t: t + dur, ch: a.ch, key: k });
        t += dur;
    }
    events.push(Event::End { t });

    write_events_to_stdout(&events)
}

use anyhow::Result;
use clap::Parser;
use music_ir::{read_events_from_stdin, write_events_to_stdout, Event};

#[derive(Parser, Debug)]
struct Args {
    /// Semitones, can be negative
    #[arg(long)]
    semitones: i16,
}

fn clamp_u8(v: i16) -> u8 {
    v.clamp(0, 127) as u8
}

fn main() -> Result<()> {
    let a = Args::parse();
    let mut events = read_events_from_stdin()?;

    for ev in &mut events {
        match ev {
            Event::NoteOn { key, .. } => {
                *key = clamp_u8(*key as i16 + a.semitones);
            }
            Event::NoteOff { key, .. } => {
                *key = clamp_u8(*key as i16 + a.semitones);
            }
            _ => {}
        }
    }

    write_events_to_stdout(&events)
}

use anyhow::Result;
use clap::Parser;
use music_ir::{read_events_from_stdin, write_events_to_stdout, Event};
use rand::{Rng, SeedableRng};
use rand_chacha::ChaCha8Rng;

#[derive(Parser, Debug)]
struct Args {
    /// Deterministic seed
    #[arg(long, default_value_t = 42)]
    seed: u64,

    /// Max tick jitter applied to NoteOn/NoteOff (+/-)
    #[arg(long, default_value_t = 8)]
    jitter_ticks: i32,

    /// Max velocity jitter applied to NoteOn (+/-)
    #[arg(long, default_value_t = 10)]
    jitter_vel: i16,
}

fn clamp_u32_i32(base: u32, delta: i32) -> u32 {
    let v = base as i64 + delta as i64;
    v.max(0) as u32
}

fn clamp_u8_i16(base: u8, delta: i16) -> u8 {
    let v = base as i16 + delta;
    v.clamp(1, 127) as u8
}

fn main() -> Result<()> {
    let a = Args::parse();
    let mut rng = ChaCha8Rng::seed_from_u64(a.seed);

    let mut events = read_events_from_stdin()?;

    for ev in &mut events {
        match ev {
            Event::NoteOn { t, vel, .. } => {
                let dt = rng.gen_range(-a.jitter_ticks..=a.jitter_ticks);
                *t = clamp_u32_i32(*t, dt);

                let dv = rng.gen_range(-a.jitter_vel..=a.jitter_vel);
                *vel = clamp_u8_i16(*vel, dv);
            }
            Event::NoteOff { t, .. } => {
                let dt = rng.gen_range(-a.jitter_ticks..=a.jitter_ticks);
                *t = clamp_u32_i32(*t, dt);
            }
            _ => {}
        }
    }

    // Important: After jitter, events might be out of order.
    events.sort_by_key(|e| match e {
        Event::NoteOn { t, .. } => *t,
        Event::NoteOff { t, .. } => *t,
        Event::Tempo { t, .. } => *t,
        Event::End { t } => *t,
    });

    write_events_to_stdout(&events)
}

use anyhow::{Context, Result};
use clap::Parser;
use midly::{
    Format, Header, MetaMessage, MidiMessage, Smf, Timing, TrackEvent, TrackEventKind,
};
use music_ir::{read_events_from_stdin, Event};
use std::{fs, path::PathBuf};

#[derive(Parser, Debug)]
struct Args {
    /// Output MIDI path
    #[arg(long)]
    out: PathBuf,

    /// Ticks per quarter note
    #[arg(long, default_value_t = 480)]
    tpq: u16,
}

fn bpm_to_mpqn(bpm: u32) -> u32 {
    // microseconds per quarter note = 60_000_000 / bpm
    60_000_000u32 / bpm.max(1)
}

fn main() -> Result<()> {
    let a = Args::parse();
    let mut events = read_events_from_stdin()?;

    // Ensure sorted by time (good hygiene)
    events.sort_by_key(|e| match e {
        Event::NoteOn { t, .. } => *t,
        Event::NoteOff { t, .. } => *t,
        Event::Tempo { t, .. } => *t,
        Event::End { t } => *t,
    });

    let header = Header::new(Format::SingleTrack, Timing::Metrical(a.tpq.into()));
    let mut track: Vec<TrackEvent> = Vec::new();

    let mut last_t: u32 = 0;

    for ev in events {
        let t = match &ev {
            Event::NoteOn { t, .. } => *t,
            Event::NoteOff { t, .. } => *t,
            Event::Tempo { t, .. } => *t,
            Event::End { t } => *t,
        };

        let delta = t.saturating_sub(last_t);
        last_t = t;

        let kind = match ev {
            Event::Tempo { bpm, .. } => TrackEventKind::Meta(MetaMessage::Tempo(
                bpm_to_mpqn(bpm).into(),
            )),
            Event::NoteOn { ch, key, vel, .. } => {
                TrackEventKind::Midi {
                    channel: ch.into(),
                    message: MidiMessage::NoteOn { key: key.into(), vel: vel.into() },
                }
            }
            Event::NoteOff { ch, key, .. } => {
                TrackEventKind::Midi {
                    channel: ch.into(),
                    message: MidiMessage::NoteOff { key: key.into(), vel: 64.into() },
                }
            }
            Event::End { .. } => {
                // We'll emit EndOfTrack below, ignore here
                continue;
            }
        };

        track.push(TrackEvent { delta: delta.into(), kind });
    }

    // End of track marker
    track.push(TrackEvent {
        delta: 0.into(),
        kind: TrackEventKind::Meta(MetaMessage::EndOfTrack),
    });

    let smf = Smf { header, tracks: vec![track] };
    let mut buf = Vec::new();
    smf.write(&mut buf).context("failed to encode MIDI")?;

    fs::write(&a.out, buf).with_context(|| format!("failed to write {:?}", a.out))?;
    Ok(())
}

Cons

Risks

NoteOn/NoteOff pairing can break if you do wilder transforms later (solution: add note_id later).
